The 0/1 Knapsack problem is a classic optimization problem where you have a set of items, each with a weight and a value, and you want to determine the number of each item to include in a collection so that the total weight is less than or equal to a given limit and the total value is as large as possible. def knapsack(weights, values, capacity):
    """
    Solves the 0/1 Knapsack problem using Dynamic Programming.
    
    :param weights: List of weights for each item.
    :param values: List of values for each item.
    :param capacity: Maximum weight the knapsack can hold.
    :return: The maximum value that can be obtained with the given capacity.
    """
    n = len(weights)
    dp = [[0] * (capacity + 1) for _ in range(n + 1)]
    
    # Build the DP table
    for i in range(1, n + 1):
        for w in range(capacity + 1):
            if weights[i - 1] <= w:
                dp[i][w] = max(dp[i - 1][w], values[i - 1] + dp[i][w - weights[i - 1]])
            else:
                dp[i][w] = dp[i - 1][w]
    
    return dp[n][capacity]

        if current_distance > distances[current_node]:
            continue
            
        # Explore all neighbors of the current node
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight
            
            # Update the distance to the neighbor if a shorter path is found
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                heapq.heappush(priority_queue, (distance, neighbor))
                
    return distances[end]

}

from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = OrderedDict()

    def get(self, key):
        if key in self.cache:
            # Move the accessed item to the end of the cache (most recently used)
            self.cache.move_to_end(key)
            return self.cache[key]
        else:
            return -1

    def put(self, key, value):
        if key in self.cache:
            self.cache.move_to_end(key)  # Remove from old position
        elif len(self.cache) >= self.capacity:
            # Remove the least recently used item (first in the order)
            self.cache.popitem(last=False)

        self.cache[key] = value

# Example usage:

cache.put('b', 2)
