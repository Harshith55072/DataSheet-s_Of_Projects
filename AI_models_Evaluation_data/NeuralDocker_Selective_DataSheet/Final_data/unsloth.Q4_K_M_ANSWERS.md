**Q296.** def longest_palindromic(s):
    if len(s) < 2: return s

    longest = ""
    for i in range(len(s)):
        left, right = i - 1, i + 1
        while left >= 0 and right < len(s) and s[left] == s[right]:
            if longest == "":
                longest = s[left:right+1]
            else:
                longest = max(longest, s[left:right+1])

        left, right = i - 2, i + 2
        while left >= 0 and right < len(s) and s[left] == s[right]:
            if longest == "":
                longest = s[left:right+1]
            else:
                longest = max(longest, s[left:right+1])

    return longest
