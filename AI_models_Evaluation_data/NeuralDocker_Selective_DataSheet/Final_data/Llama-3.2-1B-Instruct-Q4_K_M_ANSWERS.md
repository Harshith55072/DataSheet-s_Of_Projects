        start_node: The node to start the search from.

```python
import sys
import heapq

def dijkstra(graph, start_node):
    """
    Find the shortest path from the start node to all other nodes in the graph.
    
    Args:
        graph (dict): A dictionary representing the graph, where each key is a node and its value is another dictionary with neighboring nodes as keys and distances as values.
        start_node (str): The node to start the search from.

    Returns:
        dict: A dictionary with the shortest distance from the start node to all other nodes. If there's no path to any node, it will return a default value (e.g., 0 for an empty graph).
    """
    # Initialize distances and previous nodes in the shortest path
    distances = {node: sys.maxsize for node in graph}
    previous_nodes = {node: None for node in graph}

    # Set the distance to the start node to 0
    distances[start_node] = 0

    # Create a priority queue with the start node
    priority_queue = [(0, start_node)]

    while priority_queue:
        current_distance, current_node = heapq.heappop(priority_queue)

        # If we've already found a shorter path, skip this node
        if current_distance > distances[current_node]:
            continue

        # Iterate over all neighbors of the current node
        for neighbor, weight in graph[current_node].items():
            distance = current_distance + weight

            # If this path is shorter than our previous best, update it
            if distance < distances[neighbor]:
                distances[neighbor] = distance
                previous_nodes[neighbor] = current_node
                heapq.heappush(priority_queue, (distance, neighbor))

    return distances

from collections import defaultdict

def dfs(graph, start_node):
    """
    Performs a Depth-First Search (DFS) traversal of the given graph.

    Args:
        graph: A dictionary representing the adjacency list of the graph.
            Each key is a node and its corresponding value is another dictionary
            where each key is an adjacent node and its value is True if it's connected,
            False otherwise.
        start_node: The node to begin the traversal from.

    Returns:
        A list of nodes visited during the traversal, in the order they were visited.
    """
    visited = set()
    traversal_order = []

    def dfs_helper(node):
        # Mark the current node as visited
        visited.add(node)
        
        # Add the current node to the traversal order
        traversal_order.append(node)

        # Recursively visit all adjacent nodes that haven't been visited yet
        for neighbor in graph.get(node, []):
            if neighbor not in visited:
                dfs_helper(neighbor)

    # Start the DFS traversal from the given start node
    dfs_helper(start_node)

    return traversal_order

# Example usage:

    'C': [],
    'D': [],
    'E': []
}

print("DFS Traversal Order:", traversal_order)

# Output:
# DFS Traversal Order: ['A', 'B', 'D', 'E', 'C']
```

This function uses a recursive helper function `dfs_helper` to perform the actual DFS traversal. The `visited` set keeps track of nodes that have already been visited, and the `traversal_order` list stores the order in which they were visited.

    """
    Find the longest palindromic substring in a given string.

    Args:
        s (str): The input string.

    """

            left -= 1
            right += 1
        return s[left + 1:right]

        # Odd length palindrome
        palindrome = expand_around_center(i, i)
        if len(palindrome) > len(longest):
            longest = palindrome

        # Even length palindrome
        palindrome = expand_around_center(i, i + 1)
        if len(palindrome) > len(longest):
            longest = palindrome

    return longest

**Q300.** Here's a Python class that implements an LRU cache with get and put operations:
```python
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity: int):
        """
        Initialize the LRU cache.

        Args:
            capacity (int): The maximum number of items in the cache.
        """
        self.capacity = capacity
        self.cache = OrderedDict()

    def get(self, key: str) -> str:
        """
        Retrieve a value from the cache.

        Args:
            key (str): The key to retrieve.

        Returns:
            str: The retrieved value if it exists in the cache.
        """
        if key not in self.cache:
            return None
        else:
            # Move the accessed item to the end of the ordered dictionary
            del self.cache[key]
            self.cache[key] = None  # Mark as recently used
            return self.cache.get(key)

    def put(self, key: str, value: str) -> None:
        """
        Add or update a value in the cache.

        Args:
            key (str): The key to add or update.
            value (str): The value to add or update.
        """
        if key in self.cache:
            del self.cache[key]
        elif len(self.cache) >= self.capacity:
            # Remove the least recently used item
            self.cache.popitem(last=False)
        else:
            self.cache[key] = value

# Example usage:
cache = LRUCache(2)  # Create an LRU cache with capacity 2

cache.put('key1', 'value1')
print(cache.get('key1'))  # Output: value1
cache.put('key2', 'value2')
print(cache.get('key2'))  # Output: value2
```
This implementation uses the `OrderedDict` class from the `collections` module to store the cache. The `get` method moves the accessed item to the end of the ordered dictionary, and the `put` method removes the least recently used item if the cache is full.

Note that this implementation has a time complexity of O(1) for both get and put operations, making it efficient for large caches. However, it's worth noting that LRU caching can lead to increased memory usage due to the need to store additional metadata (e.g., timestamps).
